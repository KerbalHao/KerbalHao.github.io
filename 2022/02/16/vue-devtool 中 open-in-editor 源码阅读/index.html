    app.use

    我在一开始的 app.use 的时候就打了一个断点，当我们开始调试的时候，就会发现当 devServer 初始化的时候，我们就在其 before 钩子函数内调用了 app.use() 来添加对 '/__open-in-editor' 的监听，当监听到请求带有该字段的时候，就会调用我们的 launchEditorMiddleware 来打开对应的 vue 文件

    app.use('/__open-in-editor', launchEditorMiddleware(() => console.log(
  `To specify an editor, specify the EDITOR env variable or ` +
  `add "editor" field to your Vue project config.\n`
)))

    本人对 webpack 了解不多，也顺便学习了一下 app.use 的源码，主要是学到了对于参数的判断处理，声明的时候只接收一个参数 fn，但是基于对参数类型的判断，筛选出真正需要的函数参数。

    个人认为，这样做的好处是，在只需要传递一个函数参数的时候，不需要像 app.use(undefined,fn) 这样传递参数。

    // 它函数声明的时候只接收一个参数 fn
app.use = function use(fn) {
  var offset = 0; // 记录真正的函数的偏移量
  var path = '/';

  // default path to '/'
  // disambiguate app.use([fn])
  // 当 `typeof fn !== 'function'` 时，会将其赋值给临时变量 arg，判断 arg
  if (typeof fn !== 'function') {
    var arg = fn;

    // 当传入的第一个参数时 数组，且不为空时，将数组第一项赋值给 arg，直到其不为数组或为空
    while (Array.isArray(arg) && arg.length !== 0) {
      arg = arg[0];
    }

    // first arg is the path
    // 当 arg 也不是函数时，将其赋值给 path 作为路径
    if (typeof arg !== 'function') {
      offset = 1;
      path = fn;
    }
  }

  // 从传递的 arguments 中，根据偏移量解析出可能的所有函数
  // flatten 用于数组拍平
  var fns = flatten(slice.call(arguments, offset));

  if (fns.length === 0) {
    throw new TypeError('app.use() requires a middleware function')
  }

  // setup router
  // 路由懒加载
  this.lazyrouter();
  var router = this._router;

  // 遍历解析出来的函数
  fns.forEach(function (fn) {
    // 对于 
    if (!fn || !fn.handle || !fn.set) {
      // 在路由插入中间件, 鉴于能力不够，暂时未深入了解
      return router.use(path, fn);
    }
    // 下面是对于 express 框架相关的处理，此处暂时不理
    // ...
  }, this);

  return this;
};

    // 就像我们之前看的 app.use 一样，这里的参数也是做了处理，若接收到的参数为函数，则赋值给 onErrorCallback，并将原来的变量赋值为 undefined

    // 好处？可以直接传入一个函数参数，而不需要对其他参数传 undefined ？
module.exports = (specifiedEditor, srcRoot, onErrorCallback) => {
  if (typeof specifiedEditor === 'function') {
    onErrorCallback = specifiedEditor
    specifiedEditor = undefined
  }

  if (typeof srcRoot === 'function') {
    onErrorCallback = srcRoot
    srcRoot = undefined
  }
  // 获取文件所在的根路径
  srcRoot = srcRoot || process.cwd()
  // 返回一个函数，真正的打开文件的核心操作就在这里了
  return function launchEditorMiddleware (req, res, next) {

  }

    到现在为止，我们在启动服务的时候 open-in-editor 的相关代码已经走了一遍了，接下来，我们来看看事件触发时的代码是如何执行的

    launchEditorMiddleware

    // 接上文，当我们点击打开文件的时候，就会执行这一段代码
// 为什么呢？因为 app.use 传递了一个 '/__open-in-editor'， 当我们点击的时候，也就发送了一个带有这一路径的请求，监听到之后，就会触发中间件的执行
function launchEditorMiddleware (req, res, next) {
    // 获取我们要打开的文件路径
    const { file } = url.parse(req.url, true).query || {}
    if (!file) {
      res.statusCode = 500
      res.end(`launch-editor-middleware: required query param "file" is missing.`)
    } else {
      // 执行打开文件操作(需要完整的文件路径，指定的要使用的编辑器，错误处理函数)
      launch(path.resolve(srcRoot, file), specifiedEditor, onErrorCallback)
      // 中间件执行结束
      res.end()
    }
  }

    launchEditor

    接下来让我们看看核心代码 launchEditor

    let _childProcess = null // 为 _childProcess 设置一个初始值
function launchEditor (file, specifiedEditor, onErrorCallback) {
  // 获取文件信息（文件名、行号和列号）
  const parsed = parseFile(file)
  let { fileName } = parsed
  const { lineNumber, columnNumber } = parsed

  // 判断是否存在该文件
  if (!fs.existsSync(fileName)) {
    return
  }

  // 还是对于参数的判断
  if (typeof specifiedEditor === 'function') {
    onErrorCallback = specifiedEditor
    specifiedEditor = undefined
  }
  // 对于错误处理函数的二次包装，可以在内部进行其他处理，我们可以稍后学习一下 wrapErrorCallback 的处理
  onErrorCallback = wrapErrorCallback(onErrorCallback)

  // 获取到要使用的编辑器路径和其他选项，这个函数主要是通过 node 方法获取到当前使用的编辑器来处理，这一步我们也稍后再来学习
  const [editor, ...args] = guessEditor(specifiedEditor)
  if (!editor) {
    onErrorCallback(fileName, null)
    return
  }
  // 因本人只有 windows 系统，其他系统暂不考虑

  /* if (
  //   process.platform === 'linux' &&
  //   fileName.startsWith('/mnt/') &&
  //   /Microsoft/i.test(os.release())
  // ) {
    // Assume WSL / "Bash on Ubuntu on Windows" is being used, and
    // that the file exists on the Windows file system.
    // `os.release()` is "4.4.0-43-Microsoft" in the current release
    // build of WSL, see: https://github.com/Microsoft/BashOnWindows/issues/423#issuecomment-221627364
    // When a Windows editor is specified, interop functionality can
    // handle the path translation, but only if a relative path is used.

    // fileName = path.relative('', fileName)
   }
*/
  // 若存在行号，则去获取其他的位置信息，并将其添加在 guessEditor 返回的 args 中
  // 不存在则直接将文件名添加到 args 中
  if (lineNumber) {
    const extraArgs = getArgumentsForPosition(editor, fileName, lineNumber, columnNumber)
    args.push.apply(args, extraArgs)
  } else {
    args.push(fileName)
  }

  // 若 _childProcess 已存在且编辑器是一个终端编辑器(vim等)
  // 则直接杀掉进程，防止存在多个终端编辑器导致迷惑
  if (_childProcess && isTerminalEditor(editor)) {
    // There's an existing editor process already and it's attached
    // to the terminal, so go kill it. Otherwise two separate editor
    // instances attach to the stdin/stdout which gets confusing.
    _childProcess.kill('SIGKILL')
  }

  // 当当前平台是 windows 平台，调用 childProcess.spawn 唤起编辑器打开代码文件
  if (process.platform === 'win32') {
    // On Windows, launch the editor in a shell because spawn can only
    // launch .exe files.
    _childProcess = childProcess.spawn(
      'cmd.exe',
      ['/C', editor].concat(args),
      { stdio: 'inherit' }
    )
  } else {
    // 暂不考虑非 windows 平台
    // _childProcess = childProcess.spawn(editor, args, { stdio: 'inherit' })
  }

  // 监听关闭和错误
  _childProcess.on('exit', function (errorCode) {
    _childProcess = null

    if (errorCode) {
      onErrorCallback(fileName, '(code ' + errorCode + ')')
    }
  })

  _childProcess.on('error', function (error) {
    onErrorCallback(fileName, error.message)
  })
}

    wrapErrorCallback

    来看一下刚才说到的错误回调函数的装饰器

    // 错误回调函数的装饰器
function wrapErrorCallback (cb) {
  return (fileName, errorMessage) => {
    console.log(
      chalk.red('Could not open ' + path.basename(fileName) + ' in the editor.')
    )
    if (errorMessage) {
      if (errorMessage[errorMessage.length - 1] !== '.') {
        errorMessage += '.'
      }
      console.log(
        chalk.red('The editor process exited with an error: ' + errorMessage)
      )
    }
    console.log()

    // 我们可以在添加了一部分代码后，再调用回调
    if (cb) cb(fileName, errorMessage)
  }
}

    guessEditor

    下面的代码则是对将要打开的编辑器的一个猜测，如果直接传了一个字符串表示指定的编辑器，则可以直接打开该编辑器，否则，我们可以通过 node 的 childProcess 获取所有正在运行中的进程，与我们保存的常量：编辑器名称表进行匹配，找到匹配的就返回出去，给到外层函数去调用打开文件的方法

    function guessEditor (specifiedEditor) {
  if (specifiedEditor) {
    return shellQuote.parse(specifiedEditor)
  }
  // We can find out which editor is currently running by:
  // `ps x` on macOS and Linux
  // `Get-Process` on Windows
  try {
    //  mac 端，暂不深入
    if (process.platform === 'darwin') {
    //...
    } else if (process.platform === 'win32') {
      // 使用 Get-Process 获取所有子线程的名称
      const output = childProcess
        .execSync('powershell -Command "Get-Process | Select-Object Path"', {
          stdio: ['pipe', 'pipe', 'ignore']
        })
        .toString()
      const runningProcesses = output.split('\r\n')
      for (let i = 0; i < runningProcesses.length; i++) {
        // `Get-Process` sometimes returns empty lines
        if (!runningProcesses[i]) {
          continue
        }

        const fullProcessPath = runningProcesses[i].trim()
        const shortProcessName = path.basename(fullProcessPath)

        if (COMMON_EDITORS_WIN.indexOf(shortProcessName) !== -1) {
          // 找到匹配的进程名则返回
          return [fullProcessPath]
        }
      }
    } else if (process.platform === 'linux') {
      // linux 跳过
    }
  } catch (error) {
    // Ignore...
  }

  // Last resort, use old skool env vars
  // 我们也可以在环境变量中配置要打开的编辑器
  if (process.env.VISUAL) {
    return [process.env.VISUAL]
  } else if (process.env.EDITOR) {
    return [process.env.EDITOR]
  }

  return [null]
}

    总结

    这一个源码读下来，收获很多，也发现了自己的很多不足：

    1. 对于上文说到的 app.use 和 launchEditorMiddleware 中的参数处理，基于对参数类型的判断，减少传参时的无用变量的传递，是挺妙的。不过对于用惯了静态类型语言的人以及要经常协作的团队来说，这一块或许是有点恶心？毕竟是基于 javaScript 动态类型语言的特性，参数类型可以随意改变，是否会容易出现不可预料的 bug 呢？
    2. 变量名和函数名都具有语义化的特点，可以很直观的看出功能和作用
    3. 对函数进行二次包装，可以在调用之前添加一些相关的代码，不过对于装饰器的定义需要有足够的普适性
    4. 这段代码的简易流程可以理解为：初始化 devServer -> before钩子挂载中间件，监听请求带有对应的字段时触发中间件 ->点击打开发送请求 -> 解析请求的 url -> 获取对应的文件信息 -> 用 childProcess 使用生命的编辑器或获取所有线程名称找到可以使用的编辑器 -> 调用 spawn 方法使用编辑器打开对应的文件 -> 在 childProcess 监听 exit 和 error 事件处理对应情况
    5. 对于我个人来说，主要是个人知识面不足，对于 node 相关知识还是需要再去了解一下。
